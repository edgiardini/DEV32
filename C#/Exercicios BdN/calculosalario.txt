using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ProjetoCalculodeSalario
{
    public class Program
    {
        static void Main(string[] args)
        {
            string nome;
            decimal salariobase, cargo, salariofinal;

            Console.WriteLine("Digite seu nome: ");
            nome = Console.ReadLine();

            Console.WriteLine("Informe seu cargo: ");
            Console.WriteLine("1 - Estagiário\n" +
                "2 - Analista\n" +
                "3 - Gerente\n" +
                "4 - Diretor");
            cargo = Convert.ToDecimal(Console.ReadLine());

            Console.WriteLine("Informe seu salário base: ");
            salariobase = Convert.ToDecimal(Console.ReadLine());

            switch (cargo)
            {
                case 1:
                    cargo = ((salariobase * 5) / 100);
                     break;

                case 2:
                    cargo = ((salariobase * 10) / 100);
                    break;

                case 3:
                    cargo = ((salariobase * 15) / 100);
                    break;

                case 4:
                    cargo = ((salariobase * 20) / 100);
                    break;

                default:
                    Console.WriteLine("Salário Inválido!");
                    break;

            }

            Console.WriteLine(nome);
            Console.WriteLine(cargo);
            Console.WriteLine(salariobase);
            Console.WriteLine("Seu bônus é de: " + cargo);

            salariofinal = salariobase + cargo;
            Console.WriteLine("Salário Final: " + salariofinal);



            if (salariofinal > 10000)
            {
                Console.WriteLine("Parabéns pelo seu salário!");
                    }

            else
            {
                Console.WriteLine("Continue tentando!");
                    }


            
            Console.ReadKey();


        }
    }
}
